43  std::basic_string<CharT> str(std::forward<Sentence>(s));
44 
45  std::transform(str.begin(), str.end(), str.begin(),
46  [](CharT ch) {
47  /* irrelevant cases for a given char type are removed at compile time by any decent compiler */
48  if (ch < 0 || rapidfuzz::common::to_unsigned(ch) > std::numeric_limits<uint32_t>::max()) {
49  return ch;
50  } else if (ch < 256) {
51  return static_cast<CharT>(extended_ascii_mapping[rapidfuzz::common::to_unsigned(ch)]);
52  } else {
53  // this requires sources to compiled, while the current version for C++ is header only
54  // this will be added to the C++ version later on.
55 #ifdef RAPIDFUZZ_PYTHON
56  return static_cast<CharT>(Unicode::UnicodeDefaultProcess(static_cast<uint32_t>(ch)));
57 #else
58  return ch;
59 #endif
60  }

64  std::find_if(str.begin(), str.end(), [](const CharT& ch) {return ch != ' '; }));

66  str.erase(
67  std::find_if(str.rbegin(), str.rend(), [](const CharT& ch) { return ch != ' '; }).base(),
68  str.end());
