155 template <typename Sentence1, typename Sentence2>
156 std::size_t levenshtein(const Sentence1& s1, const Sentence2& s2,
157  LevenshteinWeightTable weights = {1, 1, 1},
158  std::size_t max = std::numeric_limits<std::size_t>::max())
159 {
160  auto sentence1 = common::to_string_view(s1);
161  auto sentence2 = common::to_string_view(s2);
162 
163  if (weights.insert_cost == weights.delete_cost) {
164  /* uniform Levenshtein multiplied with the common factor */
165  if (weights.insert_cost == weights.replace_cost) {
166  return detail::levenshtein(sentence1, sentence2, max) * weights.insert_cost;
167  }
168  /*
169  * when replace_cost >= insert_cost + delete_cost no substitutions are performed
170  * therefore this can be implemented as InDel distance multiplied with the common factor
171  */
172  else if (weights.replace_cost >= weights.insert_cost + weights.delete_cost) {
173  return detail::weighted_levenshtein(sentence1, sentence2, max) * weights.insert_cost;
174  }
175  }
176 
177  return detail::generic_levenshtein(sentence1, sentence2, weights, max);
178 }

370 template <typename Sentence1, typename Sentence2>
371 std::size_t hamming(const Sentence1& s1, const Sentence2& s2,
372  std::size_t max = std::numeric_limits<std::size_t>::max())
373 {
374  auto sentence1 = common::to_string_view(s1);
375  auto sentence2 = common::to_string_view(s2);
376 
377  if (sentence1.size() != sentence2.size()) {
378  throw std::invalid_argument("s1 and s2 are not the same length.");
379  }
380 
381  std::size_t hamm = 0;
382 
383  for (std::size_t i = 0; i < sentence1.length(); i++) {
384  if (sentence1[i] != sentence2[i]) {
385  hamm++;
386  }
387  }
388 
389  return hamm > max ? (std::size_t)-1 : hamm;
390 }
