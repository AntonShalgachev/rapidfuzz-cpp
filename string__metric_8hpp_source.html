157 template <typename Sentence1, typename Sentence2>
158 std::size_t levenshtein(const Sentence1& s1, const Sentence2& s2,
159  LevenshteinWeightTable weights = {1, 1, 1},
160  std::size_t max = std::numeric_limits<std::size_t>::max())
161 {
162  auto sentence1 = common::to_string_view(s1);
163  auto sentence2 = common::to_string_view(s2);
164 
165  if (weights.insert_cost == weights.delete_cost) {
166  /* when insertions + deletions operations are free there can not be any edit distance */
167  if (weights.insert_cost == 0) {
168  return 0;
169  }
170 
171  /* uniform Levenshtein multiplied with the common factor */
172  if (weights.insert_cost == weights.replace_cost) {
173  // max can make use of the common divisor of the three weights
174  const std::size_t new_max = max / weights.insert_cost + (std::size_t)(max % weights.insert_cost != 0);
175  const std::size_t distance = detail::levenshtein(sentence1, sentence2, new_max) * weights.insert_cost;
176  return (distance <= max) ? distance : (std::size_t)-1;
177  }
178  /*
179  * when replace_cost >= insert_cost + delete_cost no substitutions are performed
180  * therefore this can be implemented as InDel distance multiplied with the common factor
181  */
182  else if (weights.replace_cost >= weights.insert_cost + weights.delete_cost) {
183  // max can make use of the common divisor of the three weights
184  const std::size_t new_max = max / weights.insert_cost + (std::size_t)(max % weights.insert_cost != 0);
185  const std::size_t distance = detail::weighted_levenshtein(sentence1, sentence2, new_max) * weights.insert_cost;
186  return (distance <= max) ? distance : (std::size_t)-1;
187  }
188  }
189 
190  return detail::generic_levenshtein(sentence1, sentence2, weights, max);
191 }

418 template <typename Sentence1, typename Sentence2>
419 std::size_t hamming(const Sentence1& s1, const Sentence2& s2,
420  std::size_t max = std::numeric_limits<std::size_t>::max())
421 {
422  auto sentence1 = common::to_string_view(s1);
423  auto sentence2 = common::to_string_view(s2);
424 
425  if (sentence1.size() != sentence2.size()) {
426  throw std::invalid_argument("s1 and s2 are not the same length.");
427  }
428 
429  std::size_t hamm = 0;
430 
431  for (std::size_t i = 0; i < sentence1.length(); i++) {
432  if (common::mixed_sign_unequal(sentence1[i], sentence2[i])) {
433  hamm++;
434  }
435  }
436 
437  return hamm > max ? (std::size_t)-1 : hamm;
438 }
